#include"UART.h"

void UART_INT(unsigned long fosc ,unsigned int bdr)
{
  *(volatile char *)0x98 |= (1<<5) |(1<<2);
  *(volatile char *)0x18 |= (1<<7);
  *(volatile char *)0x99 = ((fosc/bdr/16)-1);
  }
